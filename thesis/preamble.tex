% =============================================================================
% Shared preamble for the unified thesis.
% Loaded by main.tex. Union of packages across the three constituent papers,
% with conflicts resolved (see plan: i-think-b-is-optimized-catmull.md).
% =============================================================================

% --- Document class & geometry -----------------------------------------------
% (\documentclass declared in main.tex so that this file can be \input safely)
\usepackage[a4paper, margin=2.5cm]{geometry}

% --- Encoding ----------------------------------------------------------------
\usepackage[T1]{fontenc}
\usepackage[utf8]{inputenc}

% --- Math (union across papers) ----------------------------------------------
\usepackage{amsmath, amsthm, amsfonts, amssymb, mathtools}
\usepackage{bm, centernot, bbm}

% --- Tables / figures (union) ------------------------------------------------
\usepackage{booktabs, multirow, rotating, lscape, makecell}
\usepackage{dcolumn, float}
\usepackage{adjustbox, graphicx}
\usepackage{caption, subcaption}
\usepackage{siunitx, tabularx}

% --- Algorithms / TikZ (paper 1) ---------------------------------------------
\usepackage{algorithm, algpseudocode}
\usepackage{pgf, tikz}
\usetikzlibrary{arrows, automata}

% --- Utilities ---------------------------------------------------------------
\usepackage{csquotes, enumitem, comment}
% arydshln (needed for \hdashline in ch.1) redefines \multicolumn as a non-\long
% \def, which breaks any \multicolumn cell containing \par (e.g. the malaria
% appendix data-sources table). No thesis table uses dashed *vertical* lines
% through a \multicolumn, so we restore LaTeX's original \long \multicolumn
% after loading the package; \hdashline still works.
\let\LaTeXorigmulticolumn\multicolumn
\usepackage{arydshln}
\let\multicolumn\LaTeXorigmulticolumn
\usepackage[disable]{todonotes}   % \todo{...} from paper 1 silently rendered as nothing
\usepackage[onehalfspacing]{setspace}

% --- Bibliography: biber, per-chapter refsections ----------------------------
\usepackage[
  backend=biber,
  style=authoryear,
  citestyle=authoryear,
  natbib=true,
  useprefix=true,
  maxbibnames=99,
  mincitenames=1,
  maxcitenames=2,
  labeldate=year,
  refsection=chapter,
  bibencoding=utf8
]{biblatex}
\addbibresource{bibs/intro-conclusion.bib}
% NOTE: bibs/mendeley-intro-conclusion.bib is a RAW Mendeley export (staging
% input) and is intentionally NOT loaded here. Cleaned entries from it are
% curated into bibs/intro-conclusion.bib before being cited.
\addbibresource{bibs/00-intro.bib}
\addbibresource{bibs/01-survey-sampling.bib}
\addbibresource{bibs/02-malaria.bib}
\addbibresource{bibs/03-bebbo.bib}

% --- Hyperref (load late) ----------------------------------------------------
\usepackage{hyperref}
\hypersetup{
  colorlinks=true,
  citecolor=black,
  linkcolor=black,
  urlcolor=blue,
  pdftitle={Representativity and Ecological Validity in Online Studies -- Nandan Rao},
  pdfauthor={Nandan Rao}
}

% --- Math operators (from paper 1) -------------------------------------------
\DeclareMathOperator*{\argmax}{argmax}
\DeclareMathOperator*{\argmin}{argmin}

% --- Theorems (paper 1) ------------------------------------------------------
\theoremstyle{plain}
\newtheorem{proposition}{Proposition}[chapter]

% --- Custom column types (papers 1 & 3) --------------------------------------
\newcolumntype{Y}{>{\centering\arraybackslash}X}
\newcolumntype{K}[1]{>{\centering\arraybackslash}p{#1}}
\newcolumntype{P}[1]{>{\centering\arraybackslash}p{#1}}

% --- siunitx config (consistent with vendored table inputs) ------------------
\sisetup{
  detect-mode,
  tight-spacing       = true,
  group-digits        = false,
  table-align-text-post = false
}

% --- List of Tables / List of Figures spacing ---------------------------------
% The default number box (2.3em) is too narrow for e.g. "4.10.1".
\makeatletter
\renewcommand*\l@table{\@dottedtocline{1}{1.5em}{3.6em}}
\renewcommand*\l@figure{\@dottedtocline{1}{1.5em}{3.6em}}
\makeatother

% --- Convenience macros (carry-overs from papers) ----------------------------
% Significance-star / number macros from paper 2's preamble (esttab style),
% used in body prose and table inputs (the table files \providecommand them too).
\providecommand{\num}[1]{#1}
\providecommand{\sym}[1]{\ifmmode^{#1}\else\(^{#1}\)\fi}

\providecommand{\starnote}{%
  \captionsetup{font=footnotesize}%
  \caption*{* p $<$ 0.1, ** p $<$ 0.05, *** p $<$ 0.01. Standard errors in parentheses.}%
}
